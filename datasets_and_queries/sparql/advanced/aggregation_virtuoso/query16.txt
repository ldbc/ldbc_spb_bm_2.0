#  Query name : query16
#  Query Description : 
#  Retrieve a list of N creative works, their tags, categories and titles that contain 
#  the term "policy", ordered by their tag

# reasoning rdfs:subClassOf
# reasoning rdfs:subPropertyOf

SELECT DISTINCT ?thing ?tag ?category ?title
WHERE 
{ 
  ?thing rdf:type cwork:CreativeWork . 
  ?thing rdf:type ?class . 
  ?class rdfs:subClassOf cwork:CreativeWork . 
  ?thing cwork:tag ?tag . 
  ?thing ?a ?o . 
  ?a rdfs:subPropertyOf cwork:tag . 
  ?thing cwork:category ?category . 
  ?thing cwork:title ?title . 
  ?title bif:contains '"policy"' . 
  FILTER ((?tag = ?o)) .
  OPTIONAL {
    ?thing cwork:audience ?audience . 
    FILTER ( ?audience = {{{cwAudienceType}}} ) .
  }
}
ORDER BY ?tag
LIMIT {{{randomLimit}}}
